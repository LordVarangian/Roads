custom_button = {
	name = raid_button
	tooltip = raid_button_tooltip
	potential = {
		always = yes
	}
   frame = { 
       number = 1
       trigger = { always = yes }
   }
	trigger = {

	}
	effect = {
		set_province_flag = checked_province
		set_province_flag = original_province
		owner = {
			while = {
				limit = {
					capital_scope = {
						NOT = { has_province_flag = checked_province }
					}
					any_owned_province = {
						has_province_flag = checked_province
						any_neighbor_province = {
							NOT = { has_province_flag = checked_province }
						}
					}
				}
				every_owned_province = {
					limit = {
						has_province_flag = checked_province
					}
					every_neighbor_province = {
						limit = {
							owned_by = ROOT
							NOT = { has_province_flag = checked_province }
						}
						set_province_flag = checked_province
					}
				}
			}

			if = {
				limit = {
					capital_scope = {
						has_province_flag = checked_province
					}
				}
				random_owned_province = {
					limit = {
						has_province_flag = original_province
					}
					add_base_production = 15
				}
			}
			every_owned_province = {
				limit = {
					has_province_flag = checked_province
				}
				clr_province_flag = checked_province
			}
		}
		clr_province_flag = original_province
	}
}
